portfolio-widget

A lightweight, embeddable investment return tracker for personal websites & resumes.

v1.0 · MIT · ~6kb

Live demo — dark theme

Embed code

Drop this into any HTML page
<!-- 1. Add a container -->
<div id="my-portfolio"></div>

<!-- 2. Load the widget -->
<script src="https://cdn.jsdelivr.net/gh/YOUR_USERNAME/portfolio-widget/portfolio-widget.js"></script>

<!-- 3. Configure your holdings -->
<script>
  PortfolioWidget.init({
    container:    '#my-portfolio',
    title:        'My Investment Portfolio',
    lastUpdated:  '2024-06-01',   // update this when you update prices
    assets: [
      {
        ticker:        'AAPL',
        name:          'Apple Inc.',
        allocation:    40,              // % of total portfolio
        purchaseDate:  '2022-01-15',
        purchasePrice: 172.50,
        currentPrice:  213.00,          // update manually (v2 will auto-fetch)
      },
      {
        ticker:        'BTC',
        name:          'Bitcoin',
        allocation:    35,
        purchaseDate:  '2021-06-01',
        purchasePrice: 35000,
        currentPrice:  67000,
      },
      {
        ticker:        'VTI',
        name:          'Vanguard Total Market',
        allocation:    25,
        purchaseDate:  '2020-03-01',
        purchasePrice: 140.00,
        currentPrice:  242.00,
      },
    ]
  });
</script>

All configuration options

PortfolioWidget.init({
  // Required
  assets:       [],              // array of asset objects (see above)

  // Layout
  container:    '#my-portfolio', // CSS selector or DOM element
  title:        null,            // optional string shown above widget
  height:       220,             // chart height in px
  showStats:    true,            // show summary stat cards
  showTable:    true,            // show holdings breakdown table

  // Style
  theme:        'auto',          // 'light' | 'dark' | 'auto'
  accentColor:  null,            // override chart line hex, e.g. '#6366f1'

  // Localisation
  currency:     'USD',           // ISO 4217 code: 'EUR', 'GBP', 'JPY' …
  locale:       'en-US',         // BCP 47 locale for number/date formatting

  // Footer
  lastUpdated:  null,            // date string shown in footer, e.g. '2024-06-01'
});
